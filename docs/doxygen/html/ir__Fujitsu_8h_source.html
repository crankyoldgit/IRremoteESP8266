51 union FujitsuProtocol {
52  struct {
53  uint8_t longcode[kFujitsuAcStateLength];
54  uint8_t shortcode[kFujitsuAcStateLengthShort];
55  };
56  struct {
57  // Byte 0~4
58  uint64_t :40;
59  // Byte 5
60  uint64_t Cmd :8; // short codes:cmd; long codes:fixed value
61  // Byte 6
62  uint64_t RestLength :8; // Nr. of bytes in the message after this byte.
63  // Byte 7
64  uint64_t :8;
65  // Byte 8
66  uint64_t Power :1;
67  uint64_t :3;
68  uint64_t Temp :4;
69  // Byte 9
70  uint64_t Mode :3;
71  uint64_t Clean :1;
72  uint64_t TimerType :2;
73  uint64_t :2;
74  // Byte 10
75  uint64_t Fan :3;
76  uint64_t :1;
77  uint64_t Swing :2;
78  uint64_t :2;
79  // Byte 11~13
80  uint64_t OffTimer :11; // Also is the sleep timer value
81  uint64_t OffTimerEnable :1;
82  uint64_t OnTimer :11;
83  uint64_t OnTimerEnable :1;
84  // Byte 14
85  uint64_t :3;
86  uint64_t Filter :1;
87  uint64_t :1;
88  uint64_t unknown :1;
89  uint64_t :1;
90  uint64_t OutsideQuiet :1;
91  // Byte 15
92  uint64_t :0;
93  };
94 };
