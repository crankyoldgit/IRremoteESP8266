187 union PanasonicAc32Protocol {
188  uint32_t raw;
189  struct {
190  // Byte 0
191  uint8_t :3;
192  uint8_t SwingH :1;
193  uint8_t SwingV :3;
194  uint8_t :1;
195  // Byte 1
196  uint8_t :8; // Always seems to be 0x36.
197  // Byte 2
198  uint8_t Temp :4;
199  uint8_t Fan :4;
200  // Byte 3
201  uint8_t Mode :3;
202  uint8_t PowerToggle :1; // 0 means toggle, 1 = keep the same.
203  uint8_t :4;
204  };
205 };
