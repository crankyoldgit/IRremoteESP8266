136union CoolixProtocol {
137 uint32_t raw;
138 struct { // Only 24 bits are used.
139 // Byte
140 uint32_t :1; // Unknown
141 uint32_t ZoneFollow1:1;
142 uint32_t Mode :2;
143 uint32_t Temp :4;
144 // Byte
145 uint32_t SensorTemp :5;
146 uint32_t Fan :3;
147 // Byte
148 uint32_t :3; // Unknown
149 uint32_t ZoneFollow2:1;
150 uint32_t :4;
151 };
152};
